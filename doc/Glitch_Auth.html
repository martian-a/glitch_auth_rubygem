def initialize(token, redirect_uri)

  # unique to glitch
  @path = 'https://api.glitch.com/oauth2/authorize'
  
  # unique to client
  @token = token  
  @redirect_uri = redirect_uri

  # default option values
  response_type('code')
  scope('identity')
  state('hello-world')

end
